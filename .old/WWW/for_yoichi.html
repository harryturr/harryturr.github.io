# for x axis in terms of pi
def format_func(value, tick_number):
    # find number of multiples of pi/2
    N = int(np.round(2 * value / np.pi))
    if N == 0:
        return "0"
    elif N == 1:
        return r"$\pi/2$"
    elif N == 2:
        return r"$\pi$"
    elif N % 2 > 0:
        return r"${0}\pi/2$".format(N)
    else:
        return r"${0}\pi$".format(N // 2)

x_norm = [x_norm[i] * xrange for i in range(0, len(x_norm))]

# normalizing y axis (amplitude of reflectance)
y_norm = [(y[i] - np.amin(y)) / (np.amax(y) - np.amin(y)) for i in range(0, len(y))]

# plotting the theory from equation 9
fig, ax = plt.subplots();

N = 500
phi = np.linspace(0, 8 * np.pi, N)
R1 = 0.3
R2 = 0.96

alpha = 0.5 * ((R1 + R2) / (2 * np.sqrt(R1 * R2))-1)
F = 4 * np.sqrt(R1 * R2) / (1 - np.sqrt(R1 * R2))**2
refl = [F * (alpha + np.sin(x / 20 **2)) / (1 + F * (np.sin(x / 2)**2)) for x in phi]

ax.plot(phi,refl)
ax.set_title('new eq 9 (R1 = 0.3 and R2 = 0.96)')
ax.set_ylabel('intensity $I_r / I_i$')
ax.set_xlabel('phase delay $\phi$')
plt.show()
  • the peaks are pointed upwards
    • we expect in transmission mode (at least in R1 = R2 case)
  • typically in reflective mode the peaks are down
    • (as in our data).
